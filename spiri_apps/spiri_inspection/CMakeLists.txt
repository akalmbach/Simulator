
cmake_minimum_required(VERSION 2.8.3)
project(spiri_inspection)

find_package(catkin REQUIRED COMPONENTS spiri_api)
find_package(PythonLibs 2.7 REQUIRED)
include_directories(${catkin_INCLUDE_DIRS} ${PYTHON_INCLUDE_DIRS})
add_executable(inspect inspect.cpp)
target_link_libraries(inspect ${catkin_LIBRARIES} ${Boost_LIBRARIES} ${PYTHON_LIBRARIES})
